    FILES

        ${ttkWrapperSourceFiles}
)

    PUBLIC
    FILE_SET HEADERS
    TYPE HEADERS
    BASE_DIRS

        ${CMAKE_CURRENT_SOURCE_DIR}/generated
    FILES
        ${ttkWrapperHeaders}
    PRIVATE
        ${ttkWrapperSourceFiles}
)
